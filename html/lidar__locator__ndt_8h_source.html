41 void color_mapping(float value, float midvalue, unsigned char* r,
42  unsigned char* g, unsigned char* b) {
43  if (value > 1.f) {
44  value = 1.f;
45  } else if (value < 0.f) {
46  value = 0.f;
47  }
48  if (value > midvalue) {
49  value = (value - midvalue) / (1.f - midvalue);
50  *r = value * 255.0;
51  *g = (1.0 - value) * 255.0;
52  *b = 0.0;
53  } else {
54  value /= midvalue;
55  *r = 0.0;
56  *g = value * 255.0;
57  *b = (1 - value) * 255.0;
58  }
